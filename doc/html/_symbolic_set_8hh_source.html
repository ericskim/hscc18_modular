98  SymbolicSet(const SymbolicSet& set1, const SymbolicSet& set2) {
99  m_dim = set1.m_dim + set2.m_dim;
100  m_eta.reset(new double[m_dim]);
101  m_first.reset(new double[m_dim]);
102  m_no_grid_points.reset(new abs_type[m_dim]);
103  m_NN.reset(new abs_type[m_dim]);
104  for(int i=0; i<set1.m_dim; i++) {
105  m_eta[i] = set1.m_eta[i];

107  m_no_grid_points[i] = set1.m_no_grid_points[i];
108  m_bdd_interval.push_back(set1.m_bdd_interval[i]);
109  }
110  for(int j=set1.m_dim, i=0; i<set2.m_dim; i++) {
111  m_eta[j+i] = set2.m_eta[i];
112  m_first[j+i] = set2.m_first[i];
113  m_no_grid_points[j+i] = set2.m_no_grid_points[i];

137  BDD id_to_bdd(abs_type id) const {
138  abs_type num;
139  int k=m_dim-1;
140  /* k= m_dim -1 */
141  num=id/m_NN[k];
142  id=id%m_NN[k];
143  BDD bdd = m_bdd_interval[k].int_to_bdd(num);

145  num=id/m_NN[k];
146  id=id%m_NN[k];
147  bdd = bdd & m_bdd_interval[k].int_to_bdd(num);
148  }
149  return bdd;

195  std::vector<double> bdd_to_grid_points(const Cudd& manager, BDD bdd) {
196  if((!get_no_bdd_vars()) || bdd==manager.bddZero())
197  return {};
198  /* disable reordering (if enabled) */
199  Cudd_ReorderingType *method=nullptr;
200  if(manager.ReorderingStatus(method))
201  manager.AutodynDisable();
202  /* get variable ids */
203  auto var_id = get_bdd_var_ids();
204  /* find the variables in the support of the BDD but outside the SymbolicSet */
205  auto support_id = bdd.SupportIndices();
206  std::vector<BDD> out{};
207  for(const auto& id : support_id) {
208  if(std::find(std::begin(var_id), std::end(var_id), id)==std::end(var_id))
209  out.emplace_back(manager.bddVar(id));
210  }
211  /* remove those variables from the bdd */
212  if(out.size())
213  bdd = bdd.ExistAbstract(manager.computeCube(out));
214  /* limit the grid points in the grid */
215  for(const auto& interval : m_bdd_interval)
216  bdd = bdd & interval.get_all_elements();
217  /* init the vector of grid points to be returned */
218  abs_type no_gp = get_size(manager,bdd);
219  std::vector<double> gp(no_gp*m_dim);
220  for(abs_type i=0; i<no_gp; i++) {
221  for(int j=0; j<m_dim; j++)
222  gp[i*m_dim+j]=m_first[j];
223  }
224  /* set up iteration to iterate over BDD cubes */
225  DdManager* dd = manager.getManager();
226  int *cube;
227  CUDD_VALUE_TYPE value;
228  DdGen *gen;
229  abs_type counter=0;
230  /* iterate over BDD cubes */
231  Cudd_ForeachCube(dd,bdd.getNode(),gen,cube,value) {
232  abs_type offset=1;
233  for(int i=0; i<m_dim; i++) {
234  unsigned int no_vars = m_bdd_interval[i].get_no_bdd_vars();
235  for (unsigned int j=0; j<no_vars; j++) {
236  unsigned int id = m_bdd_interval[i].get_bdd_var_ids()[j];
237  if(cube[id]==1) {
238  for(abs_type k=0; k<offset; k++) {
239  gp[(counter+k)*m_dim+i]+=(abs_type{1}<<(no_vars-1-j))*m_eta[i];
240  }
241  }
242  /* take care of don't care */
243  if(cube[id]==2) {
244  for(abs_type k=0; k<offset; k++) {
245  for(int l=0; l<=i; l++) {
246  gp[(counter+k+offset)*m_dim+l]=gp[(counter+k)*m_dim+l];
247  }
248  }
249  for(abs_type k=0; k<offset; k++) {
250  gp[(counter+k+offset)*m_dim+i]+=(abs_type{1}<<(no_vars-1-j))*m_eta[i];
251  }
252  offset=(offset<<1);
253  }
254  }
255  }
256  counter+=offset;
257  }
258  /* reactivate reordering if it was enabled */
259  if(method!=nullptr)
260  manager.AutodynEnable(*method);
261  return gp;
262  }

305  std::vector<double> restriction(const Cudd& manager,
306  const BDD& bdd,
307  const grid_point_t& x,
308  std::vector<int> domain = {}) const {
309  /* compute indices in domain/codomain */
310  std::vector<int> codomain {};
311  /* fill in default values */
312  if(domain.size()==0) {
313  domain.resize(x.size());
314  std::iota(std::begin(domain),std::end(domain),0);
315  codomain.resize(m_dim-x.size());
316  std::iota(std::begin(codomain),std::end(codomain),x.size());
317  } else {
318  for(int i=0; i<m_dim; i++) {
319  if(std::find(std::begin(domain), std::end(domain), i) == std::end(domain))
320  codomain.push_back(i);
321  }
322  }
323  /* create SymbolicSet of the domain and codmain */
324  SymbolicSet set_dom(*this,domain);
325  SymbolicSet set_codom(*this,codomain);
326  /* extract bdd with restriction */
327  abs_type id = set_dom.xtoi(x);
328  BDD restricted = bdd & set_dom.id_to_bdd(id);
329  restricted = restricted.ExistAbstract(set_dom.get_cube(manager));
330  /* map restricted BDD to grid points */
331  return set_codom.bdd_to_grid_points(manager,restricted);
332  }
