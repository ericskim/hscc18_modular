97  m_dim = set1.m_dim + set2.m_dim;
98  m_eta = new double[m_dim];
99  m_first = new double[m_dim];
100  m_no_grid_points = new abs_type[m_dim];
101  m_NN = new abs_type[m_dim];
102  for(int i=0; i<set1.m_dim; i++) {

105  m_no_grid_points[i] = set1.m_no_grid_points[i];
106  m_bdd_interval.push_back(set1.m_bdd_interval[i]);
107  }
108  for(int j=set1.m_dim, i=0; i<set2.m_dim; i++) {
109  m_eta[j+i] = set2.m_eta[i];
110  m_first[j+i] = set2.m_first[i];
111  m_no_grid_points[j+i] = set2.m_no_grid_points[i];
112  m_bdd_interval.push_back(set2.m_bdd_interval[i]);
113  }

135  BDD id_to_bdd(abs_type id) const {
136  BDD bdd;
137  abs_type num;
138  int k=m_dim-1;
139  /* k= m_dim -1 */
140  num=id/m_NN[k];
141  id=id%m_NN[k];
142  bdd = m_bdd_interval[k].int_to_bdd(num);
143  for(k=m_dim-2; k >= 0; k--) {
144  num=id/m_NN[k];
145  id=id%m_NN[k];
146  bdd = bdd & m_bdd_interval[k].int_to_bdd(num);
147  }
148  return bdd;

184  std::vector<double> bdd_to_grid_points(const Cudd& manager, BDD bdd) {
185  if((!get_no_bdd_vars()) || bdd==manager.bddZero())
186  return {};
187  /* disable reordering (if enabled) */
188  Cudd_ReorderingType *method=nullptr;
189  if(manager.ReorderingStatus(method))
190  manager.AutodynDisable();
191  /* get variable ids */
192  auto var_id = get_bdd_var_ids();
193  /* find the variables in the support of the BDD but outside the SymbolicSet */
194  auto support_id = bdd.SupportIndices();
195  std::vector<BDD> out{};
196  for(const auto& id : support_id) {
197  if(std::find(var_id.begin(), var_id.end(), id)==var_id.end())
198  out.emplace_back(manager.bddVar(id));
199  }
200  /* remove those variables from the bdd */
201  if(out.size())
202  bdd = bdd.ExistAbstract(manager.computeCube(out));
203  /* limit the grid points in the grid */
204  for(const auto& interval : m_bdd_interval)
205  bdd = bdd & interval.get_all_elements();
206  /* init the vector of grid points to be returned */
207  std::vector<double> gp(get_size(bdd)*m_dim);
208  for(abs_type i=0; i<get_size(bdd); i++) {
209  for(int j=0; j<m_dim; j++)
210  gp[i*m_dim+j]=m_first[j];
211  }
212  /* set up iteration to iterate over BDD cubes */
213  DdManager* dd = manager.getManager();
214  int *cube;
215  CUDD_VALUE_TYPE value;
216  DdGen *gen;
217  abs_type counter=0;
218  /* iterate over BDD cubes */
219  Cudd_ForeachCube(dd,bdd.getNode(),gen,cube,value) {
220  abs_type offset=1;
221  for(int i=0; i<m_dim; i++) {
222  unsigned int no_vars = m_bdd_interval[i].get_no_bdd_vars();
223  for (unsigned int j=0; j<no_vars; j++) {
224  unsigned int id = m_bdd_interval[i].get_bdd_var_ids()[j];
225  if(cube[id]==1) {
226  gp[counter*m_dim+i]+=(abs_type{1}<<(no_vars-1-j))*m_eta[i];
227  }
228  /* take care of don't care */
229  if(cube[id]==2) {
230  for(abs_type k=0; k<offset; k++) {
231  for(int l=0; l<=i; l++) {
232  gp[(counter+k+offset)*m_dim+l]=gp[(counter+k)*m_dim+l];
233  }
234  }
235  for(abs_type k=0; k<offset; k++) {
236  gp[(counter+k+offset)*m_dim+i]+=(abs_type{1}<<(no_vars-1-j))*m_eta[i];
237  }
238  offset=(offset<<1);
239  }
240  }
241  }
242  counter+=offset;
243  }
244  /* reactivate reordering if it was enabled */
245  if(method!=nullptr)
246  manager.AutodynEnable(*method);
247  return gp;
248  }

291  std::vector<double> restriction(const Cudd& manager,
292  const BDD& bdd,
293  const grid_point_t& x,
294  std::vector<int> domain = {}) const {
295  /* compute indices in domain/codomain */
296  std::vector<int> codomain {};
297  /* fill in default values */
298  if(domain.size()==0) {
299  domain.resize(x.size());
300  std::iota(domain.begin(),domain.end(),0);
301  codomain.resize(m_dim-x.size());
302  std::iota(codomain.begin(),codomain.end(),x.size());
303  } else {
304  for(int i=0; i<m_dim; i++) {
305  if(std::find(domain.begin(), domain.end(), i) == domain.end())
306  codomain.push_back(i);
307  }
308  }
309  /* create SymbolicSet of the domain and codmain */
310  SymbolicSet set_dom(*this,domain);
311  SymbolicSet set_codom(*this,codomain);
312  /* extract bdd with restriction */
313  abs_type id = set_dom.xtoi(x);
314  BDD restricted = bdd & set_dom.id_to_bdd(id);
315  restricted = restricted.ExistAbstract(manager.computeCube(set_dom.get_bdd_vars()));
316  restricted.PrintMinterm();
317  /* map restricted BDD to grid points */
318  return set_codom.bdd_to_grid_points(manager,restricted);
319  }
