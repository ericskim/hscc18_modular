47  int carry = 1;
48  for(int i=num-1; carry && (i>=0) ;i--) {
49  if(phase[i]) {
50  phase[i]=0;
51  carry=1;

53  phase[i]=1;
54  carry=0;
55  }

71  IntegerInterval(const Cudd& manager,
72  int_type lb, int_type ub,
73  const std::vector<unsigned int>& bdd_var_id = {}) :
74  m_lb(lb), m_ub(ub), m_bdd_var_id(bdd_var_id) {
75  /* compute necessar no of BDD variables to represent the integer in the interval */
76  m_size = ub-lb+1;
77  unsigned int no_bdd_var = m_bdd_var_id.size();
78  if(!no_bdd_var) {
79  int_type x = ub-lb;

82  no_bdd_var++;
83  }
84  if(!no_bdd_var) {
85  no_bdd_var=1;
86  }
87  }
88  /* get new BDD variable IDs */
89  m_bdd_vars.resize(no_bdd_var);
90  m_bdd_var_id.resize(no_bdd_var);
91  std::unique_ptr<BDD[]> vars{ new BDD[no_bdd_var] };
92  for(unsigned int j=0; j<no_bdd_var; j++) {
93  /* copy ids from bdd_var_id or create new ones */
94  if(!bdd_var_id.size()) {
95  /* get new IDs */
96  vars[j] = manager.bddVar();
97  m_bdd_var_id[j]=vars[j].NodeReadIndex();
98  } else {
99  /* copy IDs from bdd_var_id */
100  vars[j] = manager.bddVar(bdd_var_id[j]);
101  }
102  m_bdd_vars[j]=vars[j];
103  }
104  /* create BDDs to represent the integers in the interval */
105  std::unique_ptr<int[]> phase{ new int[no_bdd_var] () };
106  m_int_to_bdd.resize(m_size);
107  for(int_type i=0; i<m_size; i++) {
108  m_int_to_bdd[i] = manager.bddComputeCube(vars.get(),phase.get(),no_bdd_var);
109  add_one(phase.get(),no_bdd_var);
110  //m_int_to_bdd[i].PrintMinterm();
111  }
112  }
