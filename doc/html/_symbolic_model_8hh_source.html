69  bool post_interval_bounds(int post_dim, state_type x, state_type r, state_type eta, state_type ll, state_type ur, abs_type &post_lb, abs_type &post_ub){
70  double left = x[post_dim]-r[post_dim]-m_z[post_dim];
71  double right = x[post_dim]+r[post_dim]+m_z[post_dim];
72  if(left <= ll[post_dim]-eta[post_dim]/2.0 || right >= ur[post_dim]+eta[post_dim]/2.0) {
73  return false;
74  }
75  /* integer coordinate of lower left corner of post */
76  post_lb = static_cast<abs_type>((left-ll[post_dim]+eta[post_dim]/2.0)/eta[post_dim]);
77  /* integer coordinate of upper right corner of post */
78  post_ub = static_cast<abs_type>((right-ll[post_dim]+eta[post_dim]/2.0)/eta[post_dim]);
79  return true;
80  }

162  /* radius of hyper interval containing the attainable set */
163  state_type eta, r;

167  input_type u;
168 
169  /* Used as default value for system/radius post when iterating over lower dimensions*/
170  const std::vector<double> center_x = m_pre.get_center();
171  const std::vector<double> center_u = m_input.get_center();
172 
173  /* for out of bounds check */
174  state_type lower_left, upper_right;
175 
176  /* variables for managing the low dimensional post */
177  abs_type post_lb, post_ub;
178 
179  /* copy data from m_state_alphabet */
180  for(int i=0; i<dim; i++) {
181  eta[i]=m_pre.get_eta()[i];
182  lower_left[i]=m_pre.get_lower_left()[i];
183  upper_right[i]=m_pre.get_upper_right()[i];
184  }
185 
186  /* the BDD to encode the transition function */
187  BDD tf = manager.bddOne();
188 
189  /* State and input space BDDs */
190  BDD input_grid = m_input.get_grid_bdd(manager);
191  BDD pre_grid = m_pre.get_grid_bdd(manager);
192 
193  /* Loop over post state update dimensions */
194  for(int post_dim=0; post_dim<dim; post_dim++){
195 
196  /* Get lower dimensional dependent subspaces */
197  SymbolicSet dep_pre = SymbolicSet(m_pre, sys_dep.get_state_dependency(post_dim));
198  SymbolicSet dep_input = SymbolicSet(m_input, sys_dep.get_input_dependency(post_dim));
199  SymbolicSet dep_post = SymbolicSet(m_post, {post_dim});
200  proj_dim = dep_pre.get_dim();
201  proj_i_dim = dep_input.get_dim();
202 
203  /*Grid sizes among dependent subspaces*/
204  abs_type N = dep_pre.size();
205  abs_type M = dep_input.size();
206 
207  /*BDD to encode post_dim's coordinate update*/
208  BDD coordinate_tf = manager.bddZero();
209 
210  /* Different for loops depending on state and input dependencies*/
211  if (proj_dim > 0 && proj_i_dim > 0){ // state and input dependence
212  for(abs_type i=0; i<N; i++) { /*Loop over low dimensional pre states*/
213  BDD bdd_i = dep_pre.id_to_bdd(i); // current low dim state
214  for(abs_type j=0; j<M; j++) { /* Loop over low dimensional inputs */
215  BDD bdd_j = dep_input.id_to_bdd(j); // current low dim input
216  x = lifted_input<state_type>(i, m_pre, dep_pre, sys_dep.get_state_dependency(post_dim));
217  u = lifted_input<input_type>(j, m_input, dep_input, sys_dep.get_input_dependency(post_dim));
218 
219  /*Simulate with growth bound*/
220  for(int k=0; k<dim; k++)
221  r[k]=eta[k]/2.0+m_z[k];
222  radius_post(r,x,u);
223  system_post(x,u);
224  /* Get bounds in post_dim-th coordinate and skip if violation occurs */
225  if(!post_interval_bounds(post_dim, x, r, eta, lower_left, upper_right, post_lb, post_ub))
226  continue;
227  /* Compute BDD of low dimensional post */
228  BDD bdd_k = dep_post.interval_to_bdd(manager,post_lb,post_ub);
229 
230  /* Add transition to current post coordinate */
231  coordinate_tf = coordinate_tf | (bdd_i & bdd_j & bdd_k);
232  }
233  }
234  }
235  else if (proj_dim == 0 && proj_i_dim > 0){ // no state dependence
236  for(abs_type j=0; j<M; j++) { /* Loop over low dimensional inputs */
237  BDD bdd_j = dep_input.id_to_bdd(j); // current low dim input
238  std::copy(center_x.begin(), center_x.end(), x.begin());
239  u = lifted_input<input_type>(j, m_input, dep_input, sys_dep.get_input_dependency(post_dim));
240 
241  /*Simulate with growth bound*/
242  for(int k=0; k<dim; k++)
243  r[k]=eta[k]/2.0+m_z[k];
244  radius_post(r,x,u);
245  system_post(x,u);
246  /* Get bounds in post_dim-th coordinate and skip if violation occurs */
247  if(!post_interval_bounds(post_dim, x, r, eta, lower_left, upper_right, post_lb, post_ub))
248  continue;
249  /* Compute BDD of low dimensional post */
250  BDD bdd_k = dep_post.interval_to_bdd(manager,post_lb,post_ub);
251 
252  /* Add transition to current post coordinate */
253  coordinate_tf = coordinate_tf | (bdd_j & pre_grid & bdd_k);
254  }
255  }
256  else if (proj_dim > 0 && proj_i_dim == 0){ // no input dependence
257  /*Loop over low dimensional pre states*/
258  for(abs_type i=0; i<N; i++) {
259  BDD bdd_i = dep_pre.id_to_bdd(i); // current low dim state
260  std::copy(center_u.begin(), center_u.end(), u.begin());
261  x = lifted_input<state_type>(i, m_pre, dep_pre, sys_dep.get_state_dependency(post_dim));
262 
263  /*Simulate with growth bound*/
264  for(int k=0; k<dim; k++)
265  r[k]=eta[k]/2.0+m_z[k];
266  radius_post(r,x,u);
267  system_post(x,u);
268  /* Get bounds in post_dim-th coordinate and skip if violation occurs */
269  if(!post_interval_bounds(post_dim, x, r, eta, lower_left, upper_right, post_lb, post_ub))
270  continue;
271  /* Compute BDD of low dimensional post */
272  BDD bdd_k = dep_post.interval_to_bdd(manager,post_lb,post_ub);
273 
274  /* Add transition to current post coordinate */
275  coordinate_tf = coordinate_tf | (bdd_i & input_grid & bdd_k);
276  }
277  }
278  else{
279  throw std::runtime_error("scots::SymbolicModel A post state update does not depend on any pre state or input");
280  }
281 
282  /* Impose coordinate constraint on post_dim*/
283  tf = tf & coordinate_tf;
284 
285  } // end post state loop
286 
287  /* count number of transitions */
288  size_t nvars = m_pre.get_no_bdd_vars() +
289  m_input.get_no_bdd_vars() +
290  m_post.get_no_bdd_vars();
291  no_trans=tf.CountMinterm(nvars);
292  return tf;

311  template<class F1, class F2, class F3>
312  BDD compute_gb(const Cudd& manager, F1& system_post, F2& radius_post, F3&& avoid, size_t& no_trans) {
313  /* number of cells */
314  abs_type N=m_pre.size();
315  /* number of inputs */
316  abs_type M=m_input.size();
317  /* state space dimension */
318  int dim=m_pre.get_dim();
319  /* for display purpose */
320  abs_type counter=0;
321  /* variables for managing the post */
322  std::vector<abs_type> lb(dim); /* lower-left corner */
323  std::vector<abs_type> ub(dim); /* upper-right corner */
324  /* radius of hyper interval containing the attainable set */
325  state_type eta;
326  state_type r;
327  /* state and input variables */
328  state_type x;
329  input_type u;
330  /* for out of bounds check */
331  state_type lower_left;
332  state_type upper_right;
333  /* copy data from m_state_alphabet */
334  for(int i=0; i<dim; i++) {
335  eta[i]=m_pre.get_eta()[i];
336  lower_left[i]=m_pre.get_lower_left()[i];
337  upper_right[i]=m_pre.get_upper_right()[i];
338  }
339  /* the BDD to encode the transition function */
340  BDD tf = manager.bddZero();
341  /* is post of (i,j) out of domain ? */
342  bool out_of_domain;
343  /* loop over all cells */
344  for(abs_type i=0; i<N; i++) {
345  BDD bdd_i = m_pre.id_to_bdd(i);
346  /* is i an element of the avoid symbols ? */
347  if(avoid(i)) {
348  continue;
349  }
350  /* loop over all inputs */
351  for(abs_type j=0; j<M; j++) {
352  BDD bdd_j = m_input.id_to_bdd(j);
353  /* get center x of cell */
354  m_pre.itox(i,x);
355  /* cell radius (including measurement errors) */
356  for(int k=0; k<dim; k++)
357  r[k]=eta[k]/2.0+m_z[k];
358  /* current input */
359  m_input.itox(j,u);
360  /* integrate system and radius growth bound */
361  /* the result is stored in x and r */
362  radius_post(r,x,u);
363  system_post(x,u);
364  /* determine the cells which intersect with the attainable set:
365  * discrete hyper interval of cell indices
366  * [lb[0]; ub[0]] x .... x [lb[dim-1]; ub[dim-1]]
367  * covers attainable set */
368  for(int k=0; k<dim; k++) {
369  /* check for out of bounds */
370  double left = x[k]-r[k]-m_z[k];
371  double right = x[k]+r[k]+m_z[k];
372  if(left <= lower_left[k]-eta[k]/2.0 || right >= upper_right[k]+eta[k]/2.0) {
373  out_of_domain=true;
374  break;
375  }
376  /* integer coordinate of lower left corner of post */
377  lb[k] = static_cast<abs_type>((left-lower_left[k]+eta[k]/2.0)/eta[k]);
378  /* integer coordinate of upper right corner of post */
379  ub[k] = static_cast<abs_type>((right-lower_left[k]+eta[k]/2.0)/eta[k]);
380  }
381  if(out_of_domain) {
382  out_of_domain=false;
383  continue;
384  }
385  /* compute BDD of post */
386  BDD bdd_k = m_post.interval_to_bdd(manager,lb,ub);
387  /* add to transition function */
388  tf = tf | (bdd_i & bdd_j & bdd_k);
389  }
390  /* print progress */
391  progress(i,N,counter);
392  }
393 
394  /* count number of transitions */
395  size_t nvars = m_pre.get_no_bdd_vars() +
396  m_input.get_no_bdd_vars() +
397  m_post.get_no_bdd_vars();
398  no_trans=tf.CountMinterm(nvars);
399  return tf;
400  }
