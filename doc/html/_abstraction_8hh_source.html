134  template<class F1, class F2, class F3=decltype(params::avoid_abs)>
135  void compute_gb(TransitionFunction& transition_function,
136  F1& system_post,
137  F2& radius_post,
138  F3& avoid=params::avoid_abs) {
139  /* number of cells */
140  abs_type N=m_state_alphabet.size();
141  /* number of inputs */
142  abs_type M=m_input_alphabet.size();
143  /* number of transitions (to be computed) */
144  abs_ptr_type T=0;
145  /* state space dimension */
146  int dim=m_state_alphabet.get_dim();
147  /* for display purpose */
148  abs_type counter=0;
149  /* some grid information */
150  std::vector<abs_type> NN=m_state_alphabet.get_nn();
151  /* variables for managing the post */
152  std::vector<abs_type> lb(dim); /* lower-left corner */
153  std::vector<abs_type> ub(dim); /* upper-right corner */
154  std::vector<abs_type> no(dim); /* number of cells per dim */
155  std::vector<abs_type> cc(dim); /* coordinate of current cell in the post */
156  /* radius of hyper interval containing the attainable set */
157  state_type eta;
158  state_type r;
159  /* state and input variables */
160  state_type x;
161  input_type u;
162  /* for out of bounds check */
163  state_type lower_left;
164  state_type upper_right;
165  /* copy data from m_state_alphabet */
166  for(int i=0; i<dim; i++) {
167  eta[i]=m_state_alphabet.get_eta()[i];
168  lower_left[i]=m_state_alphabet.get_lower_left()[i];
169  upper_right[i]=m_state_alphabet.get_upper_right()[i];
170  }
171  /* init in transition_function the members no_pre, no_post, pre_ptr */
172  transition_function.init_infrastructure(N,M);
173  /* lower-left & upper-right corners of hyper rectangle of cells that cover attainable set */
174  std::unique_ptr<abs_type[]> corner_IDs(new abs_type[N*M*2]());
175  /* is post of (i,j) out of domain ? */
176  std::unique_ptr<bool[]> out_of_domain(new bool[N*M]());
177  /*
178  * first loop: compute corner_IDs:
179  * corner_IDs[i*M+j][0] = lower-left cell index of over-approximation of attainable set
180  * corner_IDs[i*M+j][1] = upper-right cell index of over-approximation of attainable set
181  */
182  /* loop over all cells */
183  for(abs_type i=0; i<N; i++) {
184  /* is i an element of the avoid symbols ? */
185  if(avoid(i)) {
186  for(abs_type j=0; j<M; j++) {
187  out_of_domain[i*M+j]=true;
188  }
189  continue;
190  }
191  /* loop over all inputs */
192  for(abs_type j=0; j<M; j++) {
193  out_of_domain[i*M+j]=false;
194  /* get center x of cell */
195  m_state_alphabet.itox(i,x);
196  /* cell radius (including measurement errors) */
197  for(int k=0; k<dim; k++)
198  r[k]=eta[k]/2.0+m_z[k];
199  /* current input */
200  m_input_alphabet.itox(j,u);
201  /* integrate system and radius growth bound */
202  /* the result is stored in x and r */
203  radius_post(r,x,u);
204  system_post(x,u);
205  /* determine the cells which intersect with the attainable set:
206  * discrete hyper interval of cell indices
207  * [lb[0]; ub[0]] x .... x [lb[dim-1]; ub[dim-1]]
208  * covers attainable set
209  */
210  abs_type npost=1;
211  for(int k=0; k<dim; k++) {
212  /* check for out of bounds */
213  double left = x[k]-r[k]-m_z[k];
214  double right = x[k]+r[k]+m_z[k];
215  if(left <= lower_left[k]-eta[k]/2.0 || right >= upper_right[k]+eta[k]/2.0) {
216  out_of_domain[i*M+j]=true;
217  break;
218  }
219 
220  /* integer coordinate of lower left corner of post */
221  lb[k] = static_cast<abs_type>((left-lower_left[k]+eta[k]/2.0)/eta[k]);
222  /* integer coordinate of upper right corner of post */
223  ub[k] = static_cast<abs_type>((right-lower_left[k]+eta[k]/2.0)/eta[k]);
224  /* number of grid points in the post in each dimension */
225  no[k]=(ub[k]-lb[k]+1);
226  /* total number of post */
227  npost*=no[k];
228  cc[k]=0;
229  }
230  corner_IDs[i*(2*M)+2*j]=0;
231  corner_IDs[i*(2*M)+2*j+1]=0;
232  if(out_of_domain[i*M+j])
233  continue;
234 
235  /* compute indices of post */
236  for(abs_type k=0; k<npost; k++) {
237  abs_type q=0;
238  for(int l=0; l<dim; l++)
239  q+=(lb[l]+cc[l])*NN[l];
240  cc[0]++;
241  for(int l=0; l<dim-1; l++) {
242  if(cc[l]==no[l]) {
243  cc[l]=0;
244  cc[l+1]++;
245  }
246  }
247  /* (i,j,q) is a transition */
248  /* increment number of pres for (q,j) */
249  transition_function.m_no_pre[q*M+j]++;
250  /* store id's of lower-left and upper-right cell */
251  if(k==0)
252  corner_IDs[i*(2*M)+2*j]=q;
253  if(k==npost-1)
254  corner_IDs[i*(2*M)+2*j+1]=q;
255  }
256  /* increment number of transitions by number of post */
257  T+=npost;
258  transition_function.m_no_post[i*M+j]=npost;
259  }
260  /* print progress */
261  if(m_verbose) {
262  if(counter==0)
263  std::cout << "1st loop: ";
264  }
265  progress(i,N,counter);
266  }
267  /* compute pre_ptr */
268  abs_ptr_type sum=0;
269  for(abs_type i=0; i<N; i++) {
270  for(abs_type j=0; j<M; j++) {
271  sum+=transition_function.m_no_pre[i*M+j];
272  transition_function.m_pre_ptr[i*M+j]=sum;
273  }
274  }
275  /* allocate memory for pre list */
276  transition_function.init_transitions(T);
277 
278  /* second loop: fill pre array */
279  counter=0;
280  for(abs_type i=0; i<N; i++) {
281  /* loop over all inputs */
282  for(abs_type j=0; j<M; j++) {
283  /* is x an element of the overflow symbols ? */
284  if(out_of_domain[i*M+j])
285  continue;
286  /* extract lower-left and upper-bound points */
287  abs_type k_lb=corner_IDs[i*2*M+2*j];
288  abs_type k_ub=corner_IDs[i*2*M+2*j+1];
289  abs_type npost=1;
290 
291  /* cell idx to coordinates */
292  for(int k=dim-1; k>=0; k--) {
293  /* integer coordinate of lower left corner */
294  lb[k]=k_lb/NN[k];
295  k_lb=k_lb-lb[k]*NN[k];
296  /* integer coordinate of upper right corner */
297  ub[k]=k_ub/NN[k];
298  k_ub=k_ub-ub[k]*NN[k];
299  /* number of grid points in each dimension in the post */
300  no[k]=(ub[k]-lb[k]+1);
301  /* total no of post of (i,j) */
302  npost*=no[k];
303  cc[k]=0;
304 
305  }
306 
307  for(abs_type k=0; k<npost; k++) {
308  abs_type q=0;
309  for(int l=0; l<dim; l++)
310  q+=(lb[l]+cc[l])*NN[l];
311  cc[0]++;
312  for(int l=0; l<dim-1; l++) {
313  if(cc[l]==no[l]) {
314  cc[l]=0;
315  cc[l+1]++;
316  }
317  }
318  /* (i,j,q) is a transition */
319  transition_function.m_pre[--transition_function.m_pre_ptr[q*M+j]]=i;
320  }
321  }
322  /* print progress */
323  if(m_verbose) {
324  if(counter==0)
325  std::cout << "2nd loop: ";
326  }
327  progress(i,N,counter);
328  }
329  }

340  template<class F1, class F2>
341  std::vector<state_type> get_post(F1& system_post,
342  F2& radius_post,
343  const state_type& x,
344  const input_type& u) const {
345  /* initialize return value */
346  std::vector<state_type> post {};
347  /* state space dimension */
348  int dim = m_state_alphabet.get_dim();
349  /* variables for managing the post */
350  std::vector<abs_type> lb(dim); /* lower-left corner */
351  std::vector<abs_type> ub(dim); /* upper-right corner */
352  std::vector<abs_type> no(dim); /* number of cells per dim */
353  std::vector<abs_type> cc(dim); /* coordinate of current cell in the post */
354  /* get radius */
355  state_type r;
356  state_type eta;
357  /* for out of bounds check */
358  state_type lower_left;
359  state_type upper_right;
360  /* fill in data */
361  for(int i=0; i<dim; i++) {
362  eta[i]=m_state_alphabet.get_eta()[i];
363  r[i]=m_state_alphabet.get_eta()[i]/2.0+m_z[i];
364  lower_left[i]=m_state_alphabet.get_lower_left()[i];
365  upper_right[i]=m_state_alphabet.get_upper_right()[i];
366  }
367  state_type xx=x;
368  /* compute growth bound and numerical solution of ODE */
369  radius_post(r,x,u);
370  system_post(xx,u);
371 
372  /* determine post */
373  abs_type npost=1;
374  for(int k=0; k<dim; k++) {
375  /* check for out of bounds */
376  double left = xx[k]-r[k]-m_z[k];
377  double right = xx[k]+r[k]+m_z[k];
378  if(left <= lower_left[k]-eta[k]/2.0 || right >= upper_right[k]+eta[k]/2.0) {
379  return post;
380  }
381  /* integer coordinate of lower left corner of post */
382  lb[k] = static_cast<abs_type>((left-lower_left[k]+eta[k]/2.0)/eta[k]);
383  /* integer coordinate of upper right corner of post */
384  ub[k] = static_cast<abs_type>((right-lower_left[k]+eta[k]/2.0)/eta[k]);
385  /* number of grid points in the post in each dimension */
386  no[k]=(ub[k]-lb[k]+1);
387  /* total number of post */
388  npost*=no[k];
389  cc[k]=0;
390  }
391  /* compute indices of post */
392  for(abs_type k=0; k<npost; k++) {
393  abs_type q=0;
394  for(int l=0; l<dim; l++) {
395  q+=(lb[l]+cc[l])*m_state_alphabet.get_nn()[l];
396  }
397  cc[0]++;
398  for(int l=0; l<dim-1; l++) {
399  if(cc[l]==no[l]) {
400  cc[l]=0;
401  cc[l+1]++;
402  }
403  }
404  /* (i,j,q) is a transition */
405  m_state_alphabet.itox(q,xx);
406  post.push_back(xx);
407  }
408  return post;
409  }
