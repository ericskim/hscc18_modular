152  template<class grid_point_t>
153  UniformGrid(const int dim,
154  const grid_point_t& lb,
155  const grid_point_t& ub,
156  const grid_point_t& eta) : UniformGrid() {
157  m_dim = dim;
158  if(m_dim != 0) {
159  /* check inut arguments */
160  for(int index=0; index<dim; index++) {
161  if(eta[index] <= 0)
162  throw std::runtime_error("\nscots::UniformGrid: eta must have positive entries.");
163  if(lb[index] > ub[index])
164  throw std::runtime_error("\nscots::UniformGrid: lower-left bound must be less than or equal to upper-right bound.");
165  }
166  m_eta.reset(new double[m_dim]);
167  m_first.reset(new double[m_dim]);
168  m_no_grid_points.reset(new abs_type[m_dim]);
169  m_NN.reset(new abs_type[m_dim]);
170 
171  /* determine number grid points in each dimension */
172  std::size_t no_l, no_u;
173  int sign_l, sign_u;
174  for(int index=0; index<m_dim; index++) {
175  m_eta[index] = eta[index];
176  /* rounding heuristic */
177  double delta = m_eta[index]/1E10;
178  try {
179  /* get sign */
180  sign_l = (lb[index] > 0) ? 1 : ((lb[index] < 0) ? -1 : 0);
181  /* compute number of grid points from zero to lower bound */
182  no_l=std::llround(std::abs(lb[index])/eta[index]+sign_l*(0.5-delta));
183  } catch (...) {
184  std::ostringstream os;
185  os << "\nscots::UniformGrid: something wrong in the division of " << lb[index] << " by " << eta[index] ;
186  throw std::runtime_error(os.str().c_str());
187  }
188  try {
189  /* get sign */
190  sign_u = (ub[index] > 0) ? 1 : ((ub[index] < 0) ? -1 : 0);
191  /* compute number of grid points from zero to upper bound */
192  no_u=std::llround(std::abs(ub[index])/eta[index]-sign_u*(0.5-delta));
193  } catch (...) {
194  std::ostringstream os;
195  os << "\nscots::UniformGrid: something wrong in the division of " << ub[index] << " by " << eta[index] ;
196  throw std::runtime_error(os.str().c_str());
197  }
198  /* check if number of grid points in dimension index does not exceed max representable by abs_type */
199  if((sign_u*no_u-sign_l*no_l+1) > std::numeric_limits<abs_type>::max()) {
200  std::ostringstream os;
201  throw std::runtime_error("\nscots::UniformGrid: number of grid points exceeds maximum value of abs_type (defined in TransitionSystem.hh).");
202  }
203  m_no_grid_points[index] = sign_u*no_u-sign_l*no_l+1;
204  /* first grid point coordinates */
205  m_first[index]= (double)sign_l*(double)no_l*eta[index];
206  }
207  calc_nn();
208  } else {
209  throw std::runtime_error("\nscots::UniformGrid: grid dimension has to be greater than zero (using non-default constructor)");
210  }
211  }

214  template<class grid_point_t>
215  abs_type xtoi(const grid_point_t& x) const {
216  abs_type id = 0;
217  double d_id;
218  double eta_h;
219 
220  for(int k=0; k<m_dim; k++) {
221  d_id = x[k]-m_first[k];
222  eta_h = m_eta[k]/2.0;
223 
224  if ( d_id <= -eta_h || d_id >= m_no_grid_points[k]*m_eta[k]+eta_h ) {
225  std::ostringstream os;
226  os << "\nscots::UniformGrid: state ";
227  for(int i=0; i<m_dim; i++) {
228  os << x[i] << " ";
229  }
230  os << "is outside uniform grid.";
231  throw std::runtime_error(os.str().c_str());
232  }
233  id += static_cast<abs_type>((d_id+eta_h )/m_eta[k])*m_NN[k];
234  }
235  return id;
236  }

240  void itox(abs_type id, grid_point_t& x) const {
241  /* map index id to grid point */
242  abs_type num;
243  for(int k = m_dim-1; k > 0; k--) {
244  num=id/m_NN[k];
