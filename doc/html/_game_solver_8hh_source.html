54 WinningDomain solve_reachability_game(const TransitionFunction& trans_function,
55  F1& target,
56  F2& avoid = params::avoid,
57  std::vector<double> & value = params::value ) {
58  /* size of state alphabet */
59  abs_type N=trans_function.m_no_states;
60  /* size of input alphabet */
61  abs_type M=trans_function.m_no_inputs;
62 
63  /* used to encode that a state is not in the winning domain */
64  abs_type loosing = std::numeric_limits<abs_type>::max();
65  if(M > loosing-1) {
66  throw std::runtime_error("scots::solve_reachability_game: Number of inputs exceeds maximum supported value");
67  }
68  /* win_domain[i] = j
69  * contains the input j associated with state i
70  *
71  * j = loosing if the target is not reachable from i
72  *
73  * initialize all states to loosing (win_domain[i]=loosing) */
74  std::vector<abs_type> win_domain(N,loosing);
75  /* initialize value */
76  value.resize(N,std::numeric_limits<double>::infinity());
77  /* keep track of the number of processed post */
78  std::unique_ptr<abs_type[]> K(new abs_type[N*M]);
79  /* keep track of the values (corresponds to M in Alg.2)*/
80  std::unique_ptr<double[]> edge_val(new double[N*M]);
81 
82  /* init fifo */
83  std::queue<abs_type> fifo;
84  for(abs_type i=0; i<N; i++) {
85  if(target(i) && !avoid(i)) {
86  /* states in the target set are defined as loosing state */
87  win_domain[i]=loosing;
88  /* value is zero */
89  value[i]=0;
90  /* states in the target are added to the fifo */
91  fifo.push(i);
92  }
93  for(abs_type j=0; j<M; j++) {
94  edge_val[i*M+j]=0;
95  K[i*M+j]=trans_function.m_no_post[i*M+j];
96  }
97  }
98 
99  /* main loop */
100  while(!fifo.empty()) {
101  /* get state to be processed */
102  abs_type q=fifo.front();
103  fifo.pop();
104  /* loop over each input */
105  for(abs_type j=0; j<M; j++) {
106  /* loop over pre's associated with this input */
107  for(abs_ptr_type v=0; v<trans_function.m_no_pre[q*M+j]; v++) {
108  abs_type i=trans_function.m_pre[trans_function.m_pre_ptr[q*M+j]+v];
109  if(avoid(i))
110  continue;
111  /* (i,j,q) is a transition */
112  /* update the number of processed posts */
113  K[i*M+j]--;
114  /* update the max value of processed posts */
115  edge_val[i*M+j]=(edge_val[i*M+j]>=1+value[q] ? edge_val[i*M+j] : 1+value[q]);
116  /* check if for node i and input j all posts are processed */
117  if(!K[i*M+j] && value[i]>edge_val[i*M+j]) {
118  fifo.push(i);
119  value[i]=edge_val[i*M+j];
120  win_domain[i]=j;
121  }
122  } /* end loop over all pres of state i under input j */
123  } /* end loop over all input j */
124  } /* fifo is empty */
125 
126  return WinningDomain(N,M,std::move(win_domain),std::vector<bool>{},loosing);
127 }

139 WinningDomain solve_invariance_game(const TransitionFunction& trans_function, F& safe) {
140  /* size of state alphabet */
141  abs_type N=trans_function.m_no_states;
142  /* size of input alphabet */
143  abs_type M=trans_function.m_no_inputs;
144  /* used to encode that a state is not in the winning domain */
145  abs_type loosing = std::numeric_limits<abs_type>::max();
146 
147  /* valid_inputs
148  * boolean array of size N*M
149  * valid_inputs[i*M+j]=true iff input j
150  * is a valid input at state i */
151  std::vector<bool> valid_inputs(N*M,false);
152  /* no_input: keep track of the number of valid inputs.
153  * If no_val_in[i]==0, then state i is not winning and
154  * no_val_in[i] is set to loosing (see also WinningDomain) */
155  std::vector<abs_type> no_val_in(N,0);
156  /* keep track if an unsafe state was already added to the fifo */
157  std::vector<bool> added(N,false);
158 
159  /* initialization */
160  std::queue<abs_type> fifo;
161  for(abs_type i=0; i<N; i++) {
162  if(safe(i)) {
163  for(abs_type j=0; j<M; j++) {
164  if(trans_function.m_no_post[i*M+j]) {
165  valid_inputs[i*M+j]=true;
166  no_val_in[i]++;
167  }
168  }
169  }
170  if(!no_val_in[i]) {
171  fifo.push(i);
172  added[i]=true;
173  /* mark no_val_in[i]=loosing to indicate that state i ist not winning */
174  no_val_in[i]=loosing;
175  }
176  }
177 
178  while(!fifo.empty()) {
179  abs_type k=fifo.front();
180  fifo.pop();
181  /* loop over all inputs */
182  for(abs_type j=0; j<M; j++) {
183  /* loop over all pre states of (k,j) */
184  for(abs_ptr_type p=0; p<trans_function.m_no_pre[k*M+j]; p++) {
185  /* (i,j,k) is a transition */
186  abs_type i=trans_function.m_pre[trans_function.m_pre_ptr[k*M+j]+p];
187  /* check if input j at state i is considered safe */
188  if(valid_inputs[i*M+j]) {
189  /* set source states i with label j as unsafe pair */
190  valid_inputs[i*M+j]=false;
191  no_val_in[i]--;
192  }
193  /* add to unsafe set if state i has no more valid inputs */
194  if(!no_val_in[i] && !added[i]) {
195  fifo.push(i);
196  added[i]=true;
197  /* mark no_val_in[i]=loosing to indicate that state i ist not winning */
198  no_val_in[i]=loosing;
199  }
200  }
201  }
202  }
203  return WinningDomain(N,M,std::move(no_val_in),std::move(valid_inputs),loosing);
204 }
