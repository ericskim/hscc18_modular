54 WinningDomain solve_reachability_game(const TransitionFunction& trans_function,
55  F1& target,
56  F2& avoid = params::avoid,
57  std::vector<double> & value = params::value ) {
58  /* size of state alphabet */
59  abs_type N=trans_function.m_no_states;
60  /* size of input alphabet */
61  abs_type M=trans_function.m_no_inputs;
62 
63  /* used to encode that a state is not in the winning domain */
64  abs_type loosing = std::numeric_limits<abs_type>::max();
65  if(M > loosing-1) {
66  throw std::runtime_error("scots::solve_reachability_game: Number of inputs exceeds maximum supported value");
67  }
68  /* win_domain[i] = j
69  * contains the input j associated with state i
70  *
71  * j = loosing if the target is not reachable from i
72  *
73  * initialize all states to loosing (win_domain[i]=loosing) */
74  std::vector<abs_type> win_domain(N,loosing);
75  /* initialize value */
76  value.resize(N,std::numeric_limits<double>::infinity());
77  /* keep track of the number of processed post */
78  std::unique_ptr<abs_type[]> K(new abs_type[N*M]);
79  /* keep track of the values (corresponds to M in Alg.2)*/
80  std::unique_ptr<double[]> edge_val(new double[N*M]);
81 
82  /* init fifo */
83  std::queue<abs_type> fifo;
84  for(abs_type i=0; i<N; i++) {
85  if(target(i) && !avoid(i)) {
86  win_domain[i]=loosing;
87  /* value is zero */
88  value[i]=0;
89  /* states in the target are added to the fifo */
90  fifo.push(i);
91  }
92  for(abs_type j=0; j<M; j++) {
93  edge_val[i*M+j]=0;
94  K[i*M+j]=trans_function.m_no_post[i*M+j];
95  }
96  }
97 
98  /* main loop */
99  while(!fifo.empty()) {
100  /* get state to be processed */
101  abs_type q=fifo.front();
102  fifo.pop();
103  /* loop over each input */
104  for(abs_type j=0; j<M; j++) {
105  /* loop over pre's associated with this input */
106  for(abs_ptr_type v=0; v<trans_function.m_no_pre[q*M+j]; v++) {
107  abs_type i=trans_function.m_pre[trans_function.m_pre_ptr[q*M+j]+v];
108  if(avoid(i))
109  continue;
110  /* (i,j,q) is a transition */
111  /* update the number of processed posts */
112  K[i*M+j]--;
113  /* update the max value of processed posts */
114  edge_val[i*M+j]=(edge_val[i*M+j]>=1+value[q] ? edge_val[i*M+j] : 1+value[q]);
115  /* check if for node i and input j all posts are processed */
116  if(!K[i*M+j] && value[i]>edge_val[i*M+j]) {
117  fifo.push(i);
118  value[i]=edge_val[i*M+j];
119  win_domain[i]=j;
120  }
121  } /* end loop over all pres of state i under input j */
122  } /* end loop over all input j */
123  } /* fifo is empty */
124 
125  return WinningDomain(N,M,std::move(win_domain),std::vector<bool>{},loosing);
126 }

137 template<class F>
138 WinningDomain solve_invariance_game(const TransitionFunction& trans_function, F& safe) {
139  /* size of state alphabet */
140  abs_type N=trans_function.m_no_states;
141  /* size of input alphabet */
142  abs_type M=trans_function.m_no_inputs;
143  /* used to encode that a state is not in the winning domain */
144  abs_type loosing = std::numeric_limits<abs_type>::max();
145 
146  /* valid_inputs
147  * boolean array of size N*M
148  * valid_inputs[i*M+j]=true iff input j
149  * is a valid input at state i */
150  std::vector<bool> valid_inputs(N*M,false);
151  /* no_input: keep track of the number of valid inputs.
152  * If no_val_in[i]==0, then state i is not winning and
153  * no_val_in[i] is set to loosing (see also WinningDomain) */
154  std::vector<abs_type> no_val_in(N,0);
155  /* keep track if an unsafe state was already added to the fifo */
156  std::vector<bool> added(N,false);
157 
158  /* initialization */
159  std::queue<abs_type> fifo;
160  for(abs_type i=0; i<N; i++) {
161  if(safe(i)) {
162  for(abs_type j=0; j<M; j++) {
163  if(trans_function.m_no_post[i*M+j]) {
164  valid_inputs[i*M+j]=true;
165  no_val_in[i]++;
166  }
167  }
168  }
169  if(!no_val_in[i]) {
170  fifo.push(i);
171  added[i]=true;
172  /* mark no_val_in[i]=loosing to indicate that state i ist not winning */
173  no_val_in[i]=loosing;
174  }
175  }
176 
177  while(!fifo.empty()) {
178  abs_type k=fifo.front();
179  fifo.pop();
180  /* loop over all inputs */
181  for(abs_type j=0; j<M; j++) {
182  /* loop over all pre states of (k,j) */
183  for(abs_ptr_type p=0; p<trans_function.m_no_pre[k*M+j]; p++) {
184  /* (i,j,k) is a transition */
185  abs_type i=trans_function.m_pre[trans_function.m_pre_ptr[k*M+j]+p];
186  /* check if input j at state i is considered safe */
187  if(valid_inputs[i*M+j]) {
188  /* set source states i with label j as unsafe pair */
189  valid_inputs[i*M+j]=false;
190  no_val_in[i]--;
191  }
192  /* add to unsafe set if state i has no more valid inputs */
193  if(!no_val_in[i] && !added[i]) {
194  fifo.push(i);
195  added[i]=true;
196  /* mark no_val_in[i]=loosing to indicate that state i ist not winning */
197  no_val_in[i]=loosing;
198  }
199  }
200  }
201  }
202  return WinningDomain(N,M,std::move(no_val_in),std::move(valid_inputs),loosing);
203 }
