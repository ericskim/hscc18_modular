43 bool is_dependent(const Cudd& mgr, const BDD& formula, const BDD& vars){
44  if (formula == mgr.bddOne() || formula == mgr.bddZero()){
45  return true;
46  }
47 
48  std::vector<unsigned int> term_support = get_support(mgr, formula);
49  std::sort(term_support.begin(), term_support.end());
50  std::vector<unsigned int> var_support = get_support(mgr, vars);
51  std::sort(var_support.begin(), var_support.end());
52  std::vector<unsigned int>::iterator iter;
53  std::vector<unsigned int> intersection_IDs(var_support.size() + term_support.size());
54  iter = std::set_intersection(term_support.begin(), term_support.end(),
55  var_support.begin(), var_support.end(),
56  intersection_IDs.begin());
57  // Check if the intersection of the support is nonempty
58  if (iter - intersection_IDs.begin() > 0){
59  return true;
60  }
61  return false;
62 }

121 class EnfPre {
122 protected:
123  /* stores the permutation array used to swap pre with post variables */
124  std::unique_ptr<int[]> m_permute;
125  /* transition relation */
126  const BDD m_tr;
127  /* transition relation with m_cube_post abstracted */
128  BDD m_tr_nopost;
129  /* BDD cubes with input and post variables */
130  BDD m_cube_post;
131  BDD m_cube_input;
132  Cudd mgr;
133 public:
142  EnfPre(const Cudd& manager,
143  const BDD& transition_relation,
144  const SymbolicSet& pre_set,
145  const SymbolicSet& control_set,
146  const SymbolicSet& post_set) : m_tr(transition_relation){
147  mgr = manager;
148  /* the permutation array */
149  size_t size = manager.ReadSize();
150  m_permute = std::unique_ptr<int[]>(new int[size]);
151  std::iota(m_permute.get(),m_permute.get()+size,0);
152  auto pre_ids = pre_set.get_bdd_var_ids();
153  auto post_ids = post_set.get_bdd_var_ids();
154  for(size_t i=0; i<pre_ids.size(); i++)
155  m_permute[pre_ids[i]]=post_ids[i];
156  /* create a cube with the input bdd vars */
157  m_cube_input = control_set.get_cube(manager);
158  /* create a cube with the post bdd vars */
159  m_cube_post = post_set.get_cube(manager);
160  /* non blocking states-input pairs */
161  m_tr_nopost=m_tr.ExistAbstract(m_cube_post);
162  }

164  template<class state_type, class input_type>
165  EnfPre(const Cudd& manager,
166  const BDD& transition_relation,
167  const SymbolicModel<state_type,input_type>& mdl): EnfPre(manager,
168  transition_relation,
169  mdl.get_sym_set_pre(),
170  mdl.get_sym_set_input(),
171  mdl.get_sym_set_post()) {
172  }

175  BDD operator()(BDD Z) const {
176  /* project onto state alphabet */
177  Z=Z.ExistAbstract(m_cube_post*m_cube_input);
178  /* swap variables */
179  Z=Z.Permute(m_permute.get());
180  /* find the (state, inputs) pairs with a post outside the safe set */
181  BDD F = m_tr.AndAbstract(!Z,m_cube_post);
182  /* the remaining (state, input) pairs make up the pre */
183  BDD preZ= m_tr_nopost & (!F);
184  return preZ;
185  }

232  /* project onto pre state alphabet */
233  Z=Z.ExistAbstract(m_cube_post*m_cube_input*m_cube_exog);
234  /* swap pre variables to post. Z is now in post domain */
235  Z=Z.Permute(m_permute.get());

237  BDD F = m_tr.AndAbstract(!Z,m_cube_post);
238  /* get rid of transitions that are inconsistent with the interconnection relation */
239  F &= inter;
240  F = F.ExistAbstract(m_cube_exog*m_cube_post);
