254 bool read_from_file(UniformGrid& grid, const std::string& filename, size_t offset = 0) {
255  FileReader reader(filename);
256  if(!reader.open()) {
257  return false;
258  }
259  int dim;
260  if(!reader.get_MEMBER(SCOTS_UG_DIM,dim,offset)) {
261  return false;
262  }
263  std::vector<double> eta;
264  if(!reader.get_VECTOR(SCOTS_UG_ETA,eta,offset)) {
265  return false;
266  }
267  std::vector<double> lb;
268  if(!reader.get_VECTOR(SCOTS_UG_LOWER_LEFT,lb,offset)) {
269  return false;
270  }
271  std::vector<double> ub;
272  if(!reader.get_VECTOR(SCOTS_UG_UPPER_RIGHT,ub,offset)) {
273  return false;
274  }
275  reader.close();
276  /* make sure that rounding in the UniformGrid constructor works correctly */
277  for(int i=0; i<dim; i++) {
278  lb[i]-=eta[i]/4.0;
279  ub[i]+=eta[i]/4.0;
280  }
281  grid = UniformGrid(dim,lb,ub,eta);
282  return true;
283 }

378 bool read_from_file(const Cudd& manager, SymbolicSet& set, const std::string& filename) {
379  /* read UniformGrid from file */
380  UniformGrid grid;
381  if(!read_from_file(grid,filename)){
382  return false;
383  }
384  /* read the BDD variable IDs and create IntegerInterval*/
385  std::vector<IntegerInterval<abs_type>> bdd_interval{};
386  FileReader reader(filename);
387  if(reader.open()) {
388  size_t offset = 0;
389  for(int i=0;i<grid.get_dim(); i++) {
390  std::vector<unsigned int> var_id {};
391  std::string s = SCOTS_SS_BDD_VAR_ID;
392  offset=reader.get_VECTOR(s.append(std::to_string(i+1)),var_id,offset);
393  if(!offset)
394  return false;
395  bdd_interval.emplace_back(manager,abs_type{0},grid.get_no_gp_per_dim()[i]-abs_type{1},var_id);
396  }
397  reader.close();
398  } else {
399  return false;
400  }
401  /* initialize SymbolicSet */
402  set = SymbolicSet(grid,bdd_interval);
403  return true;
404 }
