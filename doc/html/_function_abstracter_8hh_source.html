232  bool post_interval_bounds(int odim,

237  if(ll[odim] < (ll_bound[odim]-eta[odim]/2.0) || ur[odim] > (ur_bound[odim]+eta[odim]/2.0)){
238  std::cout << "Eta: ";
239  for (size_t i = 0; i < eta.size(); i++){
240  std::cout << eta[i] << " ";

242  std::cout << std::endl;
243  std::cout << "post_interval_bound post dim " << odim << std::endl;
244  std::cout << "LL " << ll[odim] << " " << (ll_bound[odim]-eta[odim]/2.0) << std::endl;
245  std::cout << "UR " << ur[odim] << " " << (ur_bound[odim]+eta[odim]/2.0) << std::endl;
246  return false;
247  }
248  /* integer coordinate of lower left corner of output dimension odim */
249  post_lb = static_cast<abs_type>((ll[odim]-ll_bound[odim]+eta[odim]/2.0)/eta[odim]);
250  /* integer coordinate of upper right corner of output dimension odim */
251  post_ub = static_cast<abs_type>((ur[odim]-ll_bound[odim]+eta[odim]/2.0)/eta[odim]);
252  if (post_ub < post_lb){
253  std::cout << "Output Dimension: " << odim << std::endl;
254  for (size_t i = 0; i < ll.size(); i++){
255  std::cout << ll[i] << " ";

259  std::cout << "Discrete Indices" << post_lb << " " << post_ub << std::endl;
260  throw std::runtime_error("scots::FunctionAbstracter Unexpected interval index");
261  }
262  return true;

304  const int odims = m_outSpace.get_dim();
305  if (post_dim >= odims){
306  throw std::runtime_error("scots::FunctionAbstracter post_dim exceeds permitted dimension");
307  }
308  /* variables for managing the low dimensional output */
309  concreteInput input_ll, input_ur;
310 
311  /* for out of bounds check on output */
312  concreteOutput set_ll, set_ur;
313  for(int i=0; i<odims; i++) {

315  set_ur[i]=m_outSpace.get_upper_right()[i];

324  SymbolicSet post_dim_slice = SymbolicSet(m_outSpace, {post_dim});
325 
326  abs_type N_grid_points = dep_set.size();
327  std::cout << N_grid_points << std::endl;

329  for(abs_type i=0; i<N_grid_points; i++) {
330  BDD bdd_i = dep_set.id_to_bdd(i);
331  input_ll = lifted_input<concreteInput>(i, m_inSpace, dep_set, dep.get_dependency(post_dim), "ll");
332  input_ur = lifted_input<concreteInput>(i, m_inSpace, dep_set, dep.get_dependency(post_dim), "ur");
